Add a new `delete_remote_branch` Tauri command in `src-tauri/src/commands/remote.rs`. The command accepts `path` (repo path), and `branch_name` (full remote ref name like "origin/feature-branch").

Implementation:
1. Parse `branch_name` to extract `remote` and `branch` parts by splitting on the first `/` (e.g., "origin/feature-branch" -> remote="origin", branch="feature-branch").
2. Use the existing `run_git_remote` helper to run `git push --delete --progress {remote} {branch}` with the same pattern as `git_push`.
3. After success, call `refresh_graph` to rebuild cache and emit `repo-changed`.
4. Register the command in `src-tauri/src/lib.rs` invoke_handler, right after `commands::remote::cancel_remote_op`.

```rust
#[tauri::command]
pub async fn delete_remote_branch(
    path: String,
    branch_name: String,
    state: State<'_, RepoState>,
    cache: State<'_, CommitCache>,
    running: State<'_, RunningOp>,
    app: AppHandle,
) -> Result<(), String> {
    let state_map = state.0.lock().unwrap().clone();
    let path_buf = state_map
        .get(&path)
        .ok_or_else(|| {
            serde_json::to_string(&TrunkError::new(
                "not_open",
                format!("Repository not open: {}", path),
            ))
            .unwrap()
        })?
        .clone();

    // Parse "origin/feature" into remote="origin", branch="feature"
    let slash = branch_name.find('/').ok_or_else(|| {
        serde_json::to_string(&TrunkError::new(
            "invalid_ref",
            format!("Invalid remote branch name: {}", branch_name),
        ))
        .unwrap()
    })?;
    let remote = &branch_name[..slash];
    let branch = &branch_name[slash + 1..];

    run_git_remote(
        &["push", "--delete", "--progress", remote, branch],
        &path_buf,
        &app,
        &path,
        &running.0,
    )
    .await
    .map_err(|e| serde_json::to_string(&e).unwrap())?;

    refresh_graph(&path, &state_map, &cache, &app).await
}
```

async function handleDeleteRemoteBranch(fullRefName: string) {
    const { ask } = await import("@tauri-apps/plugin-dialog");
    const confirmed = await ask(
        `Delete remote branch '${fullRefName}'? This will remove it from the remote.`,
        { title: "Delete Remote Branch", kind: "warning" },
    );
    if (!confirmed) return;
    try {
        await safeInvoke("delete_remote_branch", { path: repoPath, branchName: fullRefName });
        await loadRefs(repoPath);
        onrefreshed?.();
        showToast(`Deleted remote branch ${fullRefName}`, "success");

        const err = e as TrunkError;
        showToast(err.message ?? "Failed to delete remote branch", "error");
    }
}
